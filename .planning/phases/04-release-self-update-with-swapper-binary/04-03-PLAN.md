    ```go
    package release

    import (
        "fmt"
        "os"
        "path/filepath"
        "runtime"
    )

    //go:embed swapper/swapper_linux_amd64
    //go:embed swapper/swapper_darwin_amd64
    //go:embed swapper/swapper_darwin_arm64
    //go:embed swapper/swapper_windows_amd64.exe
    var swapperBinary []byte // Embedded compiled swapper for current platform

    // runtimeExeSuffix returns ".exe" on Windows, empty string otherwise.
    func runtimeExeSuffix() string {
        if runtime.GOOS == "windows" {
            return ".exe"
        }
        return ""
    }

    // swapperFilename returns the expected filename for the swapper binary
    // for the current GOOS/GOARCH combination.
    func swapperFilename() string {
        return fmt.Sprintf("swapper_%s_%s%s", runtime.GOOS, runtime.GOARCH, runtimeExeSuffix())
    }

    // ExtractSwapper writes the embedded swapper binary to a temp file in targetDir,
    // makes it executable, and returns the path.
    // The temp file is created in targetDir to ensure same-filesystem rename operations.
    // The caller should remove the returned file after the swapper process is spawned.
    func ExtractSwapper(targetDir string) (string, error) {
        if len(swapperBinary) == 0 {
            return "", fmt.Errorf("swapper binary not embedded for %s/%s", runtime.GOOS, runtime.GOARCH)
        }

        // Create temp file in target directory (same-filesystem for os.Rename compatibility)
        // Use a descriptive prefix so the temp file is recognizable in diagnostics
        tmpFile, err := os.CreateTemp(targetDir, ".swapper-*"+runtimeExeSuffix())
        if err != nil {
            return "", fmt.Errorf("create temp for swapper: %w", err)
        }
        tmpPath := tmpFile.Name()

        // Cleanup temp file on any error during write
        cleanup := true
        defer func() {
            if cleanup {
                tmpFile.Close()
                os.Remove(tmpPath)
            }
        }()

        if _, err := tmpFile.Write(swapperBinary); err != nil {
            return "", fmt.Errorf("write swapper binary: %w", err)
        }
        if err := tmpFile.Sync(); err != nil {
            return "", fmt.Errorf("fsync swapper binary: %w", err)
        }
        if err := tmpFile.Close(); err != nil {
            return "", fmt.Errorf("close swapper binary: %w", err)
        }

        // Make executable
        if err := os.Chmod(tmpPath, 0755); err != nil {
            return "", fmt.Errorf("chmod swapper binary: %w", err)
        }

        cleanup = false
        return tmpPath, nil
    }
    ```

    ```go
    package release

    import (
        "context"
        "errors"
        "fmt"
        "os"
        "path/filepath"
        "runtime"
    )

    // UpdateState represents the current state of a self-update operation.
    type UpdateState int

    const (
        UpdateStateUnknown  UpdateState = iota
        UpdateStateChecked              // CheckForUpdate completed
        UpdateStateDownloaded           // DownloadUpdate completed
        UpdateStateSwapperSpawned       // Swapper process started
    )

    // UpdateResult describes the outcome of a self-update attempt.
    type UpdateResult struct {
        Release     *Release    // The release that was checked/applied
        Updated     bool        // true if a new version was applied
        Current     bool        // true if already up-to-date
        State       UpdateState // How far the process got
        Err         error       // Non-nil if an error occurred
    }

    func (r UpdateResult) String() string {
        if r.Err != nil {
            return fmt.Sprintf("update failed: %v", r.Err)
        }
        if r.Current {
            return "already up-to-date"
        }
        if r.Updated {
            return fmt.Sprintf("updated to %s", r.Release.TagName)
        }
        return fmt.Sprintf("update state: %d", r.State)
    }
    ```

    ```go
    // updateLockPath returns the path to the update lock file,
    // located next to the current executable.
    func updateLockPath() (string, error) {
        exe, err := os.Executable()
        if err != nil {
            return "", err
        }
        return filepath.Dir(exe) + "/.update.lock", nil
    }

    // acquireUpdateLock creates a lock file to prevent concurrent updates.
    // Returns a cleanup function that removes the lock.
    // If the lock already exists, returns ErrUpdateInProgress.
    var ErrUpdateInProgress = errors.New("update already in progress")
    ```

    ```go
    // PerformSelfUpdate checks for an update, downloads it, extracts the swapper,
    // spawns the swapper, and exits the current process.
    //
    // If no update is available, it returns cleanly without spawning the swapper.
    // If an update is applied, this function typically does not return —
    // the calling program should os.Exit(0) after this returns.
    //
    // Usage:
    //
    //   result := release.PerformSelfUpdate(context.Background())
    //   fmt.Println(result)
    //   if result.Updated {
    //       os.Exit(0)
    //   }
    func PerformSelfUpdate(ctx context.Context, opts ...Option) UpdateResult {
        // 1. Get current version
        currentVersion, err := GetCurrentVersion()
        if err != nil {
            return UpdateResult{Err: fmt.Errorf("get current version: %w", err)}
        }

        // 2. Check for update
        release, newer, err := CheckForUpdate(ctx, currentVersion.String(), opts...)
        if err != nil {
            return UpdateResult{Err: fmt.Errorf("check update: %w", err)}
        }
        if !newer {
            return UpdateResult{Release: release, Current: true}
        }

        // 3. Acquire update lock (prevent concurrent updates)
        lockPath, err := updateLockPath()
        if err != nil {
            return UpdateResult{Err: fmt.Errorf("lock path: %w", err)}
        }
        if _, err := os.Stat(lockPath); err == nil {
            return UpdateResult{Err: ErrUpdateInProgress}
        }
        // Create lock file
        lockFile, err := os.Create(lockPath)
        if err != nil {
            return UpdateResult{Err: fmt.Errorf("create lock: %w", err)}
        }
        lockFile.Close()
        defer os.Remove(lockPath)

        // 4. Download and verify the update
        exeDir := filepath.Dir(os.Args[0])
        binaryPath, err := DownloadUpdate(ctx, release, exeDir)
        if err != nil {
            return UpdateResult{
                Release: release,
                State:   UpdateStateChecked,
                Err:     fmt.Errorf("download update: %w", err),
            }
        }
        // Compute SHA256 of downloaded binary for swapper verification
        checksumHex, err := computeFileSHA256(binaryPath)
        if err != nil {
            return UpdateResult{
                Release: release,
                State:   UpdateStateChecked,
                Err:     fmt.Errorf("compute checksum: %w", err),
            }
        }

        // 5. Extract embedded swapper to temp file
        swapperPath, err := ExtractSwapper(exeDir)
        if err != nil {
            return UpdateResult{
                Release: release,
                State:   UpdateStateDownloaded,
                Err:     fmt.Errorf("extract swapper: %w", err),
            }
        }

        // 6. Spawn swapper with --new-binary and --checksum flags
        swapperArgs := []string{
            "--new-binary=" + binaryPath,
            "--checksum=" + checksumHex,
        }
        // Append original program arguments (os.Args[1:])
        swapperArgs = append(swapperArgs, os.Args[1:]...)

        swapperProc, err := os.StartProcess(swapperPath, swapperArgs, &os.ProcAttr{
            Env:   os.Environ(),
            Files: []*os.File{os.Stdin, os.Stdout, os.Stderr},
        })
        if err != nil {
            return UpdateResult{
                Release: release,
                State:   UpdateStateDownloaded,
                Err:     fmt.Errorf("spawn swapper: %w", err),
            }
        }
        if err := swapperProc.Release(); err != nil {
            return UpdateResult{
                Release: release,
                State:   UpdateStateDownloaded,
                Err:     fmt.Errorf("release swapper: %w", err),
            }
        }

        return UpdateResult{
            Release: release,
            Updated: true,
            State:   UpdateStateSwapperSpawned,
        }
    }

    // computeFileSHA256 returns the hex-encoded SHA256 digest of the file at path.
    func computeFileSHA256(path string) (string, error) {
        data, err := os.ReadFile(path)
        if err != nil {
            return "", err
        }
        sum := sha256.Sum256(data)
        return hex.EncodeToString(sum[:]), nil
    }
    ```

    Implementation rules:

    1. **Lock file (T-04-10 mitigation)**: Create `.update.lock` next to the current executable. If it already exists, return `ErrUpdateInProgress`. Use `defer os.Remove(lockPath)` to ensure cleanup even on panic.

    2. **Checksum computation**: Before spawning the swapper, compute the SHA256 of the downloaded binary and pass it via `--checksum` flag. This is the first verification phase; the swapper will re-verify as the second phase per D-01.

    3. **Executable directory**: `exeDir` is derived from `os.Args[0]` or `os.Executable()` — use `filepath.Dir(os.Args[0])` as the primary path since `os.Executable()` may return a symlinked path. Resolve via `filepath.EvalSymlinks` if needed.

    4. **os.StartProcess**: Spawn the swapper as a separate process, then release it. The swapper will wait for a brief period for the parent to exit. The main process should `os.Exit(0)` after calling `PerformSelfUpdate`.

    5. **Return behavior**: If no update is available, return `{Current: true}` — the caller can check this and continue normal operation. If update is applied, return `{Updated: true}` — the caller should exit (the swapper will take over).

    ```go
    //

        }
    }
    ```

<threat_model>

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Compile time → Binary | Embedded swapper binary in source tree at compile time — integrity depends on build pipeline |
| Process → Lock file | File-based mutex for concurrent update prevention |
| Main process → Swapper process | Data passed via CLI flags (binary paths, checksums) |

## STRIDE Threat Register

| Threat ID | Category | Component | Severity | Disposition | Mitigation Plan |
|-----------|----------|-----------|----------|-------------|-----------------|
| T-04-10 | Tampering | PerformSelfUpdate / lock file | medium | mitigate | Acquire update lock before starting download. Lock file placed next to executable, removed on completion. `ErrUpdateInProgress` returned if lock exists. |
| T-04-11 | Tampering | go:embed swapper binary | high | mitigate | Swapper is compiled from the same source tree at build time. The Makefile target runs `go build` for each platform. CI pipeline validates source integrity via git. No runtime download of executable code per D-03. |
| T-04-12 | Information Disclosure | Lock file path | low | accept | Lock file path is deterministic (executable directory). No sensitive data in lock file — it is a zero-byte sentinel. |
| T-04-13 | Denial of Service | Orphaned lock file | low | accept | If an update crashes before removing the lock file, the next update attempt will see a stale lock and return ErrUpdateInProgress. Calling application can implement a stale-lock timeout (e.g., 5 minutes) or use a timestamp-based lock. Documented as caller responsibility. |

</threat_model>
